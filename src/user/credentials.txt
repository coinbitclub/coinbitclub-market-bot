// POST /user/credentials
export async function saveUserCredentials(req, res) {
  try {
    const { exchange, api_key, api_secret, is_testnet } = req.body;
    console.log('[🔐 CREDENTIALS BODY]', req.body);
    console.log('[🔐 USER ID]', req.user?.id);

    if (!exchange || !api_key || !api_secret || req.user?.id == null) {
      return res.status(400).json({ error: 'Dados obrigatórios ausentes.' });
    }

    const saved = await saveCredentials(req.user.id, {
      exchange,
      api_key,
      api_secret,
      is_testnet
    });

    res.json(saved);
  } catch (err) {
    console.error('[❌ ERRO SALVANDO CREDENCIAIS]', err.message);
    res.status(500).json({ error: 'Erro interno ao salvar credenciais.' });
  }
}
