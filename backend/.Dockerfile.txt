# Dockerfile
FROM node:18-alpine

# tiny init to forward signals
RUN apk add --no-cache tini

WORKDIR /app

# copie só o package.json e o lockfile primeiro (cache de dependências)
COPY backend/package*.json ./

# só dependências de produção
RUN npm ci --omit=dev

# copie todo o código do backend
COPY backend ./

# expose a porta que o app vai usar (o Railway injeta via env PORT)
EXPOSE 8080

# use tini pra repassar SIGTERM corretamente
ENTRYPOINT ["/sbin/tini", "--"]

# comando de start
CMD ["npm", "run", "start:prod"]
